# Loadstring

```lua
--// Flame's Hub 🔥\\--
getgenv().Protection_Mode = false  -- Turn "false" for afk farming
getgenv().Flames_Hub = "https://raw.githubusercontent.com/Lvl9999/Flames/main/Source";

pcall(function()
    loadstring(game:HttpGet(getgenv().Flames_Hub))();
end)
```
